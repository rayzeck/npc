	OnMobOro:

			getitem 781,1;
			set $worldboss,0;
			gvgoff "guild_vs2";
			end;

end;

	OnMobMinato:
			announce "[World Boss]: "+strcharinfo(0)+" derrotou Minato! Recompensa: 1 Fuuin no Sho!",0;
			getitem 753,1;
			set $worldboss,0;
			gvgoff "guild_vs2";
			end;

		}
