// WoE Começar/Terminar
//============================================================
-	script	Agit2_Event	-1,{
	end;

OnClock0036:	//Inicia na domingo(7) (20:30)
OnClock0100:	//Termina na domingo(7) (22:00)

OnAgitInit2:
	// Iniciando a verificação do tempo
	if((gettime(4)==7) && (gettime(3)>=00 && gettime(3)<01)) {
		if (!agitcheck2()) {
			agitstart2;
		}
		end;
	}

	// Fim da verificação do tempo
	if ((gettime(4)==7) && (gettime(3)<01)) {
		if (agitcheck2()) {
			agitend2;
		}
		end;
	}
	end;
}
